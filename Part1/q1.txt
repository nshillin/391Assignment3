PREFIX dbt: <http://dbpedia.org/resource/Template:>

SELECT ?stadium (SAMPLE(?club) AS ?clubName)
WHERE {
	?stadium dct:subject dbc:Football_venues_in_Italy .
  	?club dbp:ground ?stadium .
}
